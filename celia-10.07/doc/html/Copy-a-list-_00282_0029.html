3.3.2 Copy a list (2)

This example is the correct version of copying the data of a list into another list of different length. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 3);
  xi = NULL;
  yi = NULL;
  z = NULL;
  xi = x;
  yi = y;
  while (xi != NULL and yi != NULL)
    {
      yi->data = (xi->data);
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      z = yi->next;
      yi = NULL;
      yi = z;
      z = NULL;
    }
  return 0;
}
