3.1.1 First not null





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  int _l;
  int _k;
  int S;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  xi = x;
  while (xi != NULL and (xi->data == 0))
    {
      y = xi->next;
      xi = NULL;
      xi = y;
      y = NULL;
    }
  return 0;
}
