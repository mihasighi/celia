3.3.3 Copy a list (3)

This example is the erroneous version of copying the data of a list into another list of different length. CINV reports a dereference of a NULL pointer. The invariant generated at the end of the loop is bottom. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 3);
  xi = NULL;
  yi = NULL;
  z = NULL;
  xi = x;
  yi = y;
  while (xi != NULL)
    {
      yi->data = (xi->data);
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      z = yi->next;
      yi = NULL;
      yi = z;
      z = NULL;
    }
  return 0;
}
