3.3.6 Copy a list and add some constant (2)

This program copy the data of a list into another list by adding a constant. The two lists have different lengths, but the program correctly tests this case. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z1;
  intlist z2;
  int _l;
  int _k;
  int S;

  assume (x == 3);
  xi = NULL;
  yi = NULL;
  z1 = NULL;
  z2 = NULL;
  xi = x;
  yi = y;
  while (xi != NULL and yi != NULL)
    {
      yi->data = (xi->data + 2);
      z1 = xi->next;
      z2 = yi->next;
      xi = NULL;
      yi = NULL;
      xi = z1;
      yi = z2;
      z1 = NULL;
      z2 = NULL;
    }
  return 0;
}
