3.3.8 Insertion sort array

This version of the insertion sort algorithm does not move cells of the list but only moves data between cells. Then, it simulates the insertion sort algorithm on arrays. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  int _l;
  int _k;
  int S;
  int m;
  int n;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  yi = NULL;
  xi = x->next;
  while (xi != NULL)
    {
      y = x;
      while (y != xi and y->data <= xi->data)
	{
	  yi = y->next;
	  y = NULL;
	  y = yi;
	  yi = NULL;
	}
      m = xi->data;
      while (y != xi)
	{
	  n = y->data;
	  y->data = m;
	  m = n;
	  yi = y->next;
	  y = NULL;
	  y = yi;
	  yi = NULL;
	}
      y = NULL;
      xi->data = m;
      yi = xi->next;
      xi = NULL;
      xi = yi;
      yi = NULL;
    }
  return 0;
}
