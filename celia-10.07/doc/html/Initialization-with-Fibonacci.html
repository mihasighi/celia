3.2.5 Initialization with Fibonacci





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  int _l;
  int _k;
  int S;
  int m1;
  int m2;

  assume (x == 0);
  m1 = 1;
  m2 = 0;
  y = NULL;
  xi = NULL;
  xi = x;
  while (xi != NULL)
    {
      xi->data = (m1 + m2);
      m1 = m2;
      m2 = xi->data;
      y = xi->next;
      xi = NULL;
      xi = y;
      y = NULL;
    }
  return 0;
}
