3.4.7 Copy and reverse





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  z = NULL;
  xi = x;
  while (xi != NULL)
    {
      z = (intlist) malloc (sizeof (struct _intlist));
      z->data = (xi->data);
      z->next = y;
      y = NULL;
      y = z;
      z = NULL;
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
    }
  return 0;
}
