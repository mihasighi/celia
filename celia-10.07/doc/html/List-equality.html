3.1.4 List equality





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 2);
  xi = NULL;
  yi = NULL;
  z = NULL;
  xi = x;
  yi = y;
  while ((xi != NULL and yi != NULL and xi->data == yi->data))
    {
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      z = yi->next;
      yi = NULL;
      yi = z;
      z = NULL;
    }
  if ((xi == NULL and yi == NULL))
    {
      _k = 1;
    }
  else
    {
      _k = 0;
    }
  return 0;
}
