3.4.3 New copy on condition





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;
  int v;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  yi = NULL;
  z = NULL;
  xi = x;
  while (xi != NULL)
    {
      if ((xi->data >= v))
	{
	  yi = (intlist) malloc (sizeof (struct _intlist));
	  yi->data = (xi->data);
	  yi->next = y;
	  y = NULL;
	  y = yi;
	  yi = NULL;
	}
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
    }
  return 0;
}
