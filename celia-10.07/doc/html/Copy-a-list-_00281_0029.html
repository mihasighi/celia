3.3.1 Copy a list (1)

Copy the data of a list into another equal length list. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 2);
  xi = NULL;
  yi = NULL;
  z = NULL;
  xi = x;
  yi = y;
  while (xi != NULL)
    {
      yi->data = (xi->data);
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      z = yi->next;
      yi = NULL;
      yi = z;
      z = NULL;
    }
  return 0;
}
