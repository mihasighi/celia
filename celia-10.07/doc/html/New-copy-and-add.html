3.4.2 New copy and add





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 0);
  y = NULL;
  yi = NULL;
  z = NULL;
  xi = NULL;
  xi = x;
  while (xi != NULL)
    {
      z = (intlist) malloc (sizeof (struct _intlist));
      z->data = (xi->data + 2);
      z->next = NULL;
      if ((yi == NULL))
	{
	  y = z;
	}
      else
	{
	  yi->next = NULL;
	  yi->next = z;
	}
      yi = NULL;
      yi = z;
      z = NULL;
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
    }
  return 0;
}
