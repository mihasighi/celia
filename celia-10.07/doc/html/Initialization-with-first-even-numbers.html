3.2.3 Initialization with first even numbers





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist z;
  int _l;
  int _k;
  int S;
  int m;

  assume (x == 0);
  xi = NULL;
  z = NULL;
  m = 2;
  xi = x;
  while (xi != NULL)
    {
      xi->data = m;
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      m = m + 2;
    }
  return 0;
}
