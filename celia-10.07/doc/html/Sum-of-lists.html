3.2.7 Sum of lists





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  intlist zi;
  intlist zii;
  int _l;
  int _k;
  int S;
  int T;

  assume (x == 4);
  xi = NULL;
  yi = NULL;
  zi = NULL;
  zii = NULL;
  xi = x;
  yi = y;
  zi = z;
  while (xi != NULL and yi != NULL and zi != NULL)
    {
      zi->data = (xi->data + yi->data);
      zii = xi->next;
      xi = NULL;
      xi = zii;
      zii = NULL;
      zii = yi->next;
      yi = NULL;
      yi = zii;
      zii = NULL;
      zii = zi->next;
      zi = NULL;
      zi = zii;
      zii = NULL;
    }
  return 0;
}
