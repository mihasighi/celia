3.1.5 Sum of elements





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  int _l;
  int _k;
  int S;
  int sum;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  xi = x;
  sum = 0;
  while (xi != NULL)
    {
      sum = sum + xi->data;
      y = xi->next;
      xi = NULL;
      xi = y;
      y = NULL;
    }
  return 0;
}
