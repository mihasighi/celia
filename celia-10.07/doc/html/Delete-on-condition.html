3.4.4 Delete on condition





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist z;
  int _l;
  int _k;
  int S;
  int v;

  assume (x == 0);
  y = NULL;
  xi = NULL;
  z = NULL;
  xi = x;
  while (xi != NULL)
    {
      if ((xi->data >= v))
	{
	  z = xi;
	  xi = NULL;
	  xi = z->next;
	  if ((y == NULL))
	    {
	      x = NULL;
	      x = xi;
	    }
	  else
	    {
	      y->next = NULL;
	      y->next = xi;
	    }
	  free (z);
	  z = NULL;
	}
      else
	{
	  y = NULL;
	  y = xi;
	  z = xi->next;
	  xi = NULL;
	  xi = z;
	  z = NULL;
	}
    }
  return 0;
}
