3.3.5 Copy a list and add some constant (1)

This program copy the data of a list into another list by adding a constant. The two lists have the same length. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 2);
  xi = NULL;
  yi = NULL;
  z = NULL;
  xi = x;
  yi = y;
  while (xi != NULL)
    {
      yi->data = (xi->data + 2);
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      z = yi->next;
      yi = NULL;
      yi = z;
      z = NULL;
    }
  return 0;
}
