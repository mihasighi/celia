3.3.9 Bubble sort array





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist xin;
  int _l;
  int _k;
  int S;
  int v;

  assume (x == 0);
  xi = NULL;
  xin = NULL;
  _k = 1;
  while (_k == 1)
    {
      _k = 0;
      xi = x;
      xin = x->next;
      while (xi != NULL and xin != NULL)
	{
	  if ((xi->data >= xin->data + 1))
	    {
	      v = xi->data;
	      xi->data = (xin->data);
	      xin->data = v;
	      _k = 1;
	    }
	  xi = NULL;
	  xi = xin;
	  xin = NULL;
	  xin = xi->next;
	}
      xi = NULL;
      xin = NULL;
    }
  return 0;
}
