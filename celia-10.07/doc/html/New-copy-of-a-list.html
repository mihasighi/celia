3.4.1 New copy of a list





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 0);
  xi = NULL;
  z = NULL;
  yi = NULL;
  y = NULL;
  xi = x;
  while (xi != NULL)
    {
      z = (intlist) malloc (sizeof (struct _intlist));
      z->data = (xi->data);
      z->next = (NULL);
      if ((y == NULL))
	{
	  y = z;
	}
      else
	{
	  yi->next = z;
	}
      yi = NULL;
      yi = z;
      z = NULL;
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
    }
  return 0;
}
