3.4.6 Dispatch lists





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xgtv;
  intlist xi;
  intlist xlev;
  intlist y;
  intlist z;
  int _l;
  int _k;
  int S;
  int v;

  assume (x == 0);
  xgtv = NULL;
  xi = NULL;
  xlev = NULL;
  y = NULL;
  z = NULL;
  xi = x;
  x = NULL;
  while (xi != NULL)
    {
      y = xi;
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      y->next = NULL;
      if ((y->data <= v))
	{
	  y->next = xlev;
	  z = xlev;
	  xlev = NULL;
	  xlev = y;
	}
      else
	{
	  y->next = xgtv;
	  z = xgtv;
	  xgtv = NULL;
	  xgtv = y;
	}
      z = NULL;
      y = NULL;
    }
  return 0;
}
