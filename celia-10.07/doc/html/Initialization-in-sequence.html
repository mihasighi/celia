3.2.4 Initialization in sequence





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist z;
  int _l;
  int _k;
  int S;
  int m;
  int mp;

  assume (x == 0);
  xi = NULL;
  z = NULL;
  mp = m;
  xi = x;
  while (xi != NULL)
    {
      xi->data = mp;
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
      mp = mp + 1;
    }
  return 0;
}
