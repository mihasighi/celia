3.2.2 Initialization with first integers





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  int _l;
  int _k;
  int S;
  int m;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  m = 0;
  xi = x;
  while (xi != NULL)
    {
      xi->data = m;
      y = xi->next;
      xi = NULL;
      xi = y;
      y = NULL;
      m = m + 1;
    }
  return 0;
}
