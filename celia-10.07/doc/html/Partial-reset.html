3.2.6 Partial reset





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  intlist yi;
  int _l;
  int _k;
  int S;

  assume (x == 1);
  xi = NULL;
  yi = NULL;
  yi = y;
  while (yi != NULL)
    {
      yi->data = 0;
      xi = yi->next;
      yi = NULL;
      yi = xi;
      xi = NULL;
    }
  return 0;
}
