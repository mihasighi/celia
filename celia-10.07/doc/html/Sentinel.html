3.1.3 Sentinel

In its original version Halbwach-Peron-08, this program uses a test xi->data!=m. We have changed it below to xi->data<=m in order to avoid disequality constraints. 





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist y;
  int _l;
  int _k;
  int S;
  int m;

  assume (x == 0);
  xi = NULL;
  y = NULL;
  xi = x;
  while ((xi != NULL and xi->data <= m))
    {
      y = xi->next;
      xi = NULL;
      xi = y;
      y = NULL;
    }
  return 0;
}
