3.3.7 Set the flag





#include "intlist.h"
int
main ()
{

  intlist x;
  intlist xi;
  intlist z;
  int _l;
  int _k;
  int S;

  assume (x == 0);
  xi = NULL;
  z = NULL;
  xi = x;
  while (xi != NULL)
    {
      if ((xi->data == 0))
	{
	  xi->data = 1;
	}
      z = xi->next;
      xi = NULL;
      xi = z;
      z = NULL;
    }
  return 0;
}
